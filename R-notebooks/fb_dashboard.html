library(covidcast)
library(dplyr)
library(ggplot2)

df_state = covidcast_signal("fb-survey", "raw_cli", geo_type = "state")
n_per_week = df_state %>% 
  group_by(time_value) %>% 
  summarize(n= sum(sample_size)) 
ggplot(n_per_week, aes(x = time_value, y = n)) + 
  geom_line() + geom_point() + theme_bw() + 
  labs(x = "Date", y = "Number of Responses")

mean(n_per_week$n)
## [1] 72945.76
sum(n_per_week$n)
## [1] 9774732
df_county = covidcast_signal("fb-survey", "smoothed_cli")
county_per_week = df_county %>% 
  group_by(time_value) %>%
  summarize(n = n())
ggplot(county_per_week, aes(x = time_value, y = n)) + 
  geom_line() + geom_point() + theme_bw() + 
  labs(x = "Date", y = "Number of Counties")

mean(county_per_week$n)
